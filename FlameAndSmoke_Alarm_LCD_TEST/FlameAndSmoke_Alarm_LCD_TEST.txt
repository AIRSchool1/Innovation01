#include <Wire.h>
#include <LiquidCrystal_I2C.h>

LiquidCrystal_I2C lcd(0x27,16,2);

int smokeSensor = A0;
int flameSensor = A1;
int buzzer = 8;

int smokeValue;
int flameValue;

void setup()
{
  pinMode(buzzer, OUTPUT);

  lcd.init();
  lcd.backlight();

  Serial.begin(9600);

  lcd.setCursor(0,0);
  lcd.print("Fire & Smoke");
  lcd.setCursor(0,1);
  lcd.print("Detector");
  delay(3000);

  lcd.clear();
}

void loop()
{
  smokeValue = analogRead(smokeSensor);
  flameValue = analogRead(flameSensor);

  Serial.print("Smoke: ");
  Serial.print(smokeValue);
  Serial.print("  Flame: ");
  Serial.println(flameValue);

  lcd.setCursor(0,0);

  // Smoke level
  if(smokeValue < 300)
  {
    lcd.print("S:LOW ");
  }
  else if(smokeValue < 500)
  {
    lcd.print("S:MED ");
  }
  else
  {
    lcd.print("S:HIGH");
  }

  lcd.setCursor(8,0);
  lcd.print(smokeValue);

  lcd.setCursor(0,1);

  // Flame level
  if(flameValue > 800)
  {
    lcd.print("F:LOW ");
  }
  else if(flameValue > 500)
  {
    lcd.print("F:MED ");
  }
  else
  {
    lcd.print("F:HIGH");
  }

  lcd.setCursor(8,1);
  lcd.print(flameValue);

  // Alarm
  if(smokeValue >= 300 || flameValue <= 500)
  {
    tone(buzzer,1000);
  }
  else
  {
    noTone(buzzer);
  }

  delay(500);
}
